@app.route('/chat', methods=['POST'])
def select():
    selected_value = request.form.get('language-select')
    if selected_value == 'en':
        initial_message = "Hi!"
    elif selected_value == 'ur':
        initial_message = "!سلام"
    elif selected_value == 'bn':
        initial_message = "হাই!"
    elif selected_value == 'es':
        initial_message = "Hola!"
    elif selected_value == 'ar':
        initial_message = "!مرحبا"
    return f'Selected value: {initial_message}'



@app.route("/mp3")
def stream_mp3():
    def generate():
        with open('Answers/English/A1.mp3', "rb") as fmp3:
            data = fmp3.read(1024)
            while data:
                yield data
                data = fmp3.read(1024)
    return Response(generate(), mimetype="audio/mp3")
